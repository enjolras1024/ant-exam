请使用axios_retry发送请求，config同axios，新增配置项retry，例如：

    axios_retry({
      url: 'http://localhost:3000/data', 
      method: 'get',
      retry: {
        count: 2, // 重试次数，默认0
        sleep: 2000, // 重试前休眠时间(ms)，默认0
        maxAge: 10000 // 缓存时长(ms)，默认0。Infinity表示页面不关，缓存一直在。
      }
    })
    .then(function(res) {
      console.log(res)
    })
    .catch(function(err) {
      console.error(err)
    })
